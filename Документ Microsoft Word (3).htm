        const mobileMenu = document.getElementById('mobileMenu');

        const closeMenu = document.getElementById('closeMenu');

        const menuOverlay = document.getElementById('menuOverlay');

        const themeToggleBtn = document.getElementById('themeToggleBtn');

 

        // Переменные приложения

        let currentUser = null;

        let users = JSON.parse(localStorage.getItem('chatUsers')) || [];

        let messages = JSON.parse(localStorage.getItem('chatMessages')) || [];

        let messageCounter = messages.length;

 

        // Инициализация приложения

        function initApp() {

            // Проверяем, есть ли сохраненный пользователь

            const savedUser = localStorage.getItem('currentUser');

            if (savedUser) {

                currentUser = savedUser;

                loginModal.style.display = 'none';

                updateUserInfo();

                enableChat();

                loadMessages();

                updateOnlineUsers();

            }

           

            // Загружаем сообщения при загрузке страницы

            loadMessages();

           

            // Настройка обработчиков для мобильного меню

            setupMobileMenu();

           

            // Проверяем сохраненную тему

            const savedTheme = localStorage.getItem('chatTheme');

            if (savedTheme === 'light') {

                document.body.classList.add('light-theme');

                themeToggleBtn.innerHTML = '<i class="fas fa-sun"></i>';

            }

        }

 

        // Настройка мобильного меню

        function setupMobileMenu() {

            menuToggle.addEventListener('click', () => {

                mobileMenu.classList.add('active');

                menuOverlay.classList.add('active');

                document.body.style.overflow = 'hidden';

            });

           

            closeMenu.addEventListener('click', closeMobileMenu);

            menuOverlay.addEventListener('click', closeMobileMenu);

           

            // Закрытие меню при нажатии на ссылку

            userList.addEventListener('click', closeMobileMenu);

        }

       

        function closeMobileMenu() {

            mobileMenu.classList.remove('active');

            menuOverlay.classList.remove('active');

            document.body.style.overflow = 'auto';

        }

 

        // Функция входа в чат

        function login() {

            const username = usernameInput.value.trim();

            if (username) {

                currentUser = username;

                localStorage.setItem('currentUser', currentUser);

               

                // Добавляем пользователя в список онлайн

                if (!users.includes(username)) {

                    users.push(username);

                    localStorage.setItem('chatUsers', JSON.stringify(users));

                }

               

                loginModal.style.display = 'none';

                updateUserInfo();

                enableChat();

                updateOnlineUsers();

               

                // Показываем системное сообщение

                addSystemMessage(`${username} присоединился к чату`);

               

                showNotification(`Добро пожаловать, ${username}!`);

            }

        }

 

        // Обновление информации о пользователе

        function updateUserInfo() {

            userName.textContent = currentUser;

            userAvatar.textContent = currentUser.charAt(0).toUpperCase();

            userAvatar.style.background = getRandomColor();

        }

 

        // Включение функциональности чата

        function enableChat() {

            messageInput.disabled = false;

            sendButton.disabled = false;

        }

 

        // Отправка сообщения

        function sendMessage() {

            const text = messageInput.value.trim();

            if (text) {

                const message = {

                    id: messageCounter++,

                    sender: currentUser,

                    text: text,

                    time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})

                };

               

                // Сохраняем сообщение

                messages.push(message);

                localStorage.setItem('chatMessages', JSON.stringify(messages));

               

                // Добавляем сообщение в чат

                addMessageToChat(message, true);

               

                // Очищаем поле ввода

                messageInput.value = '';

               

                // Показываем уведомление

                showNotification('Сообщение отправлено');

               

                // Эмулируем ответ других пользователей

                simulateResponse(text);

            }

        }

 

        // Добавление сообщения в чат

        function addMessageToChat(message, isOwn = false) {

            const messageElement = document.createElement('div');

            messageElement.className = `message ${isOwn ? 'own' : 'other'}`;

           

            messageElement.innerHTML = `

                ${!isOwn ? `<div class="message-sender">${message.sender}</div>` : ''}

                <div class="message-text">${message.text}</div>

                <div class="message-time">${message.time}</div>

            `;

           

            messagesContainer.appendChild(messageElement);

            scrollToBottom();

        }

       

        // Прокрутка к последнему сообщению

        function scrollToBottom() {

            setTimeout(() => {

                messagesContainer.scrollTop = messagesContainer.scrollHeight;

            }, 100);

        }

 

        // Добавление системного сообщения

        function addSystemMessage(text) {

            const messageElement = document.createElement('div');

            messageElement.className = 'message other';

           

            messageElement.innerHTML = `

                <div class="message-sender">Система</div>

                <div class="message-text">${text}</div>

                <div class="message-time">Только что</div>

            `;

           

            messagesContainer.appendChild(messageElement);

            scrollToBottom();

           

            // Сохраняем системное сообщение

            const message = {

                id: messageCounter++,

                sender: 'Система',

                text: text,

                time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})

            };

           

            messages.push(message);

            localStorage.setItem('chatMessages', JSON.stringify(messages));

        }

 

        // Загрузка сообщений

        function loadMessages() {

            messagesContainer.innerHTML = '';

            messages.forEach(message => {

                const isOwn = message.sender === currentUser;

                addMessageToChat(message, isOwn);

            });

        }

 

        // Обновление списка онлайн-пользователей

        function updateOnlineUsers() {

            userList.innerHTML = '';

            users.forEach(user => {

                const userItem = document.createElement('li');

                userItem.className = 'user-item';

               

                userItem.innerHTML = `

                    <div class="user-status"></div>

                    <div class="user-avatar-small" style="background: ${getRandomColor()}">${user.charAt(0).toUpperCase()}</div>

                    <div class="user-name-small">${user}</div>

                `;

               

                userList.appendChild(userItem);

            });

           

            onlineCount.textContent = users.length;

        }

 

        // Эмуляция ответов других пользователей

        function simulateResponse(userMessage) {

            // Случайным образом решаем, отвечать ли на сообщение

            if (Math.random() > 0.7) {

                const responses = [

                    "Привет! Как дела?",

                    "Интересное сообщение!",

                    "Я тоже об этом думал...",

                    "Согласен с тобой!",

                    "А можно подробнее?",

                    "Не совсем понимаю, о чем ты",

                    "Это правда?",

                    "Спасибо за информацию!",

                    "Хм, нужно подумать над этим",

                    "Отличная идея!"

                ];

               

                const randomUser = users[Math.floor(Math.random() * users.length)];

                if (randomUser && randomUser !== currentUser) {

                    setTimeout(() => {

                        const response = {

                            id: messageCounter++,

                            sender: randomUser,

                            text: responses[Math.floor(Math.random() * responses.length)],

                            time: new Date().toLocaleTimeString([], {hour: '2-digit', minute:'2-digit'})

                        };

                       

                        messages.push(response);

                        localStorage.setItem('chatMessages', JSON.stringify(messages));

                        addMessageToChat(response, false);

                    }, 2000 + Math.random() * 3000);

                }

            }

        }

 

        // Показать уведомление

        function showNotification(text) {

            notificationText.textContent = text;

            notification.classList.add('show');

           

            setTimeout(() => {

                notification.classList.remove('show');

            }, 3000);

        }

 

        // Генерация случайного цвета

        function getRandomColor() {

            const colors = ['#4361ee', '#3a0ca3', '#4cc9f0', '#7209b7', '#f72585', '#4895ef'];

            return colors[Math.floor(Math.random() * colors.length)];

        }

 

        // Очистка чата

        function clearChat() {

            if (confirm('Вы уверены, что хотите очистить чат? Это действие нельзя отменить.')) {

                messages = [];

                localStorage.setItem('chatMessages', JSON.stringify(messages));

                loadMessages();

                showNotification('Чат очищен');

            }

        }

 

        // Смена темы

        function toggleTheme() {

            document.body.classList.toggle('light-theme');

            const themeIcon = document.querySelector('#themeToggleBtn i');

           

            if (document.body.classList.contains('light-theme')) {

                themeIcon.className = 'fas fa-sun';

                localStorage.setItem('chatTheme', 'light');

            } else {

                themeIcon.className = 'fas fa-moon';

                localStorage.setItem('chatTheme', 'dark');

            }

        }

        loginButton.addEventListener('click', login);

            if (e.key === 'Enter') login();

        });

       

        sendButton.addEventListener('click', sendMessage);

        messageInput.addEventListener('keypress', (e) => {

            if (e.key === 'Enter') sendMessage();

        });

       

        clearChatBtn.addEventListener('click', clearChat);

        themeToggleBtn.addEventListener('click', toggleTheme);

 

        // Обработка виртуальной клавиатуры на мобильных

        window.addEventListener('resize', scrollToBottom);

        initApp();
